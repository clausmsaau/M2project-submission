# make cleanup function 1
def cleanup(string):
  string = re.sub(r'[^\d\w.]+', ' ', string) # removes all that, aren't letters, e.g numbers.
  string = re.sub(r'[\d\_]+', ' ', string)
  return(string)

# Applying cleanup function

#Create use a counter to count the frequency of edges and assign it as weights to the newtork
c = Counter(G.edges())  # Contains frequencies of edge.

for u, v, d in G.edges(data=True):
    d['weight'] = c[u, v]#Assign c as the weight

# Calculate degre-centrality and assign it as a node-attribute
degree_centrality = nx.degree_centrality(G)
nx.set_node_attributes(G, degree_centrality, 'degree')

degree_centrality = nx.degree_centrality(H)
nx.set_node_attributes(H, degree_centrality, 'degree')
In [0]:
# Getting a "reasonable" lower bound.
perc_filter = np.percentile([v for u,v in degree_centrality.items()], 85)
In [0]:
# Make a subgraph based on nodes with a degree_centrality over the threshold
nodes_selected = [x for x,y in degree_centrality.items() if y >= perc_filter]

G = G.subgraph(nodes_selected)
H = H.subgraph(nodes_selected)
