# make cleanup function 1
def cleanup(string):
  string = re.sub(r'''(?i)\b((?:https?://|www\d{0,3}[.]|[a-z0-9.\-]+[.][a-z]{2,4}/)(?:[^\s()<>]+|\(([^\s()<>]+|(\([^\s()<>]+\)))*\))+(?:\(([^\s()<>]+|(\([^\s()<>]+\)))*\)|[^\s`!()\[\]{};:'".,<>?«»“”‘’]))''', " ", string)
  string = re.sub(r'[^\d\w.]+', ' ', string) # removes all that, aren't letters, e.g numbers.
  string = re.sub(r'[\d]+', ' ', string)
  string = re.sub(r'[\d\_]+', ' ', string)
  string = re.sub(r'[.*]+', ' ', string)
  string = re.sub(r'[*]+', ' ', string)
  string = re.sub(r'[+]+', ' ', string)
  return(string)

# Applying my cleanup function

# Lemmatize with POS Tag (if the word is a noun or ADJ etc.) "This is need to optimize the lemmatizer in the next part" 
from nltk.corpus import wordnet

def get_wordnet_pos(word):
    """Map POS tag to first character lemmatize() accepts"""
    tag = nltk.pos_tag([word])[0][1][0].upper()
    tag_dict = {"J": wordnet.ADJ,
                "N": wordnet.NOUN,
                "V": wordnet.VERB,
                "R": wordnet.ADV}

    return tag_dict.get(tag, wordnet.NOUN)

def removestopword(string):
  tokens = string.lower().split()
  final_string = [word for word in tokens if word not in stopwords_en]
  final_string = [lemmatizer.lemmatize(w, get_wordnet_pos(w)) for w in final_string]
  final_string = ' '.join(final_string)
  final_string = final_string.rstrip(punctuation)
  final_string = final_string.lstrip(punctuation)

LSA¶

LSA was introduced by Jerome Bellegarda in 2005. The purpose of LSA is reducing dimension for classification. The idea is that words will occurs in similar pieces of text if they have similar meaning. We usually use Latent Semantic Indexing (LSI) as an alternative name in NLP field.

LDA - model¶

LSI (also known as Latent Semantic Analysis, LSA) learns latent topics by performing a matrix decomposition (SVD) on the term-document matrix.

LDA is a generative probabilistic model, that assumes a Dirichlet prior over the latent topics.

In practice, LSI is much faster to train than LDA, but has lower accuracy.
